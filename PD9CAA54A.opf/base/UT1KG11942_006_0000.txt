༄༅། །བླ་མ་རྡོ་རྗེ་འཆང་གི་རྣལ་འབྱོར་སྙིང་གི་བདུད་རྩི་བཞུགས་སོ།། ན་མོ་གུ་རུ་བཛྲ་དྷ་ར་ཡ། བློ་ལྡན་གང་ཞིག་དད་པ་མོས་གུས་ལམ་གྱི་སྲོག་ཏུ་བཟུང་ནས་ངོར་ཆེན་རྡོ་རྗེ་འཆང་ལ་སྐྱབས་གནས་ཀུན་འདུས་སུ་ཤེས་པའི་སྒོ་ནས། རྣལ་འབྱོར་ཟབ་མོ་སྒོམ་པར་འདོད་པས། སངས་རྒྱས་ཆོས་ཚོགས་ལན་གསུམ་གྱིས་སྐྱབས་སེམས་སྔོན་དུ་བཏང་ནས། གདོད་ནས་རྣམ་དག་སྤྲོས་བྲལ་ངང་ལས། །གནས་ཡུལ་ཅིར་སྣང་དག་པ་རབ་འབྱམས། །བདེ་ཆེན་ཞིང་གི་བཀོད་པ་ཡོངས་རྫོགས། །མཆོད་སྤྲིན་རྒྱ་མཚོ་དཀྲིགས་པའི་དབུས་སུ། །སེང་ཁྲི་པདྨ་ཟླ་བའི་སྟེང་དུ། །དྲིན་ཅན་བླ་མ་མཆོག་དང་དབྱེར་མེད། །རྡོ་རྗེ་འཆང་དངོས་ཀུན་དགའ་བཟང་པོ། །དཀར་དམར་མདངས་ལྡན་ཞི་ཞིང་འཛུམ་པ། །ཆོས་གོས་རྣམ་གསུམ་ས་ཞུའི་ཅོད་པན། །མཛེས་པར་གསོལ་བས་དྭང་བ་བསྐྱེད་མཛད། །ཆོས་འཆད་ཕྱག་རྒྱ་ཐུགས་དཀར་མཛད་པའི། །མཐེབ་མཛུབ་བར་ནས་དགྱེས་པའི་པད་དཀར། །ཟེའུ་འབྲུའི་དབུས་གནས་རྡོ་རྗེ་དྲིལ་བུས། །ཐབས་དང་ཤེས་རབ་དབྱེར་མེད་དོན་སྟོན། །ཞབས་ཟུང་རྡོ་རྗེ་སྐྱིལ་ཀྲུང་གིས་བཞུགས། །གནས་གསུམ་འབྲུ་གསུམ་མཚན་པའི་འོད་ཀྱིས། །ཕྱོགས་བཅུའི་ཞིང་གི་རྩ་བརྒྱུད་བླ་མ། །ཡི་དམ་ལྷ་ཚོགས་རྒྱལ་བ་སྲས་བཅས། །ཆོས་སྲུང་ནོར་ལྷའི་ཚོགས་དང་བཅས་པ། །ཁ་བ་བུ་ཡུག་འཚུབ་བཞིན་སྤྱན་དྲངས། །རྗེ་བཙུན་བླ་མའི་སྐུ་ལ་ཐིམ་པས། །སྐྱབས་ཀུན་འདུས་པའི་བདག་ཉིད་ཅན་གྱུར། །དེ་ནས་སྡེ་སྣོད་མ་ཚར་གཅིག་གིས་ཡན་ལག་བདུན་པ་དང་། མཎྜལ་རྒྱས་བསྡུས་གང་འོས་སྐབས་དང་བསྟུན་པར་འབུལ། མོས་གུས་དྲག་པོས་རབ་སྡེ་རྦ་གསུམ་ཚིག་དོན་རྗེས་དྲན་གྱི་ངང་ནས་དག་གསལ་སྙན་གསུམ་དང་ལྡན་པར་བཏོན་རྗེས། མདོ་རྒྱུད་ཀུནཾ་གྱི་རྒྱ་མཚོ་ལས་བྱུང་བའི། །གསུང་ངག་ཡིད་བཞིན་ནོར་བུའི་དགཾའ་སྟོན་འགྱེད། །ཕན་བདེ་དཔལ་འབྱོར་བཟཾང་པོཾ་སྩོལ་མཛད་པ། །དཔལ་ལྡན་བླ་མའི་ཞབས་ལ་གསོལ་བ་འདེབས། །ཞེས་པ་ལ་གྲངས་བཟུང་ནས་བརྒྱ་སྟོང་སོགས་གང་མང་གསོག་མཐར།

མདུན་གྱི་བླ་མ་འོད་ཞུ་རང་ལ་ཐིམ། །རྡོ་རྗེ་གསུམ་དང་སྒོ་གསུམ་དབྱེར་མེད་ངང་། །དབང་དང་དངོས་གྲུབ་ཐམས་ཅད་ལེགས་ཐོབ་ཅིང་། །རང་སེམས་བླ་མའི་རང་ཞལ་མངོན་དུ་གྱུར། །ཞེས་བརྗོད་ནས། བླ་མའི་སྐུ་བཞིའི་ངོ་བོ་སྐྱོང་། སྐྱེ་བ་ཀུན་ཏུ་སོགས་བསྔོ་སྨོན་གྱིས་རྒྱས་བཏབ་སྟེ་ཆོས་མཐུན་གྱི་སྤྱོད་པས་དུས་འདའ་བར་བྱའོ། །ཞེས་པ་འདི་ནི་སྨད་ཤོད་རྫོང་གསར་དགོན་པའི་དགེ་སློང་ཀུན་དགའ་ཕུན་ཚོགས་ཀྱི་ཞལ་སྔ་ནས་ནན་ཏན་ཆེན་པོས་བསྐུལ་ཞིང་། ཉ་ཆེར་སྤམ་འཛང་དགོན་པའི་དགེ་སློང་མཁྱེན་རབ་ཚུལ་ཁྲིམས་ནས་ཀྱང་ཉམས་ལེན་བདེ་བའི་བླ་མའི་རྣལ་འབྱོར་ཞིག་དགོས་ཞེས་ཡང་ཡང་བསྐུལ་བ་གཉིས་ཀའི་དགོངས་པ་ཅིག་ཅར་བསྒྲུབ་པའི་ཕྱིར་དུ། རྒྱལ་བ་གཉིས་པ་རྗེ་བཙུན་ཨེ་ཝཾ་པ་ཆེན་པོའི་རྗེས་འཇུག་གི་ཚུལ་ཙམ་ཞིག་འཛིན་པ་ཤཱཀྱའི་དགེ་སློང་ངག་དབང་ལེགས་པས་དྲྭ་སྒང་སྒྲུབ་སྡེ་བསམ་གཏན་ཕུན་ཚོགས་གླིང་དུ་བྲིས་པ་དགེ་ལེགས་སུ་གྱུར་ཅིག། །། PAGE \* MERGEFORMAT 2 །སྒ་སྟོན་ངག་དབང་ལེགས་པ། བླ་མ་རྡོ་རྗེ་འཆང་གི་རྣལ་འབྱོར་སྙིང་གི་བདུད་རྩི། PAGE \* MERGEFORMAT 1
